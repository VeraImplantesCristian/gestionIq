create table public.pagos (
  id bigint generated by default as identity not null,
  created_at timestamp with time zone null default now(),
  orden_de_pago_id bigint not null,
  instrumentador_dni text not null,
  monto_total_instrumentador numeric not null,
  constraint pagos_pkey primary key (id),
  constraint pagos_instrumentador_dni_fkey foreign KEY (instrumentador_dni) references instrumentadores (dni),
  constraint pagos_orden_de_pago_id_fkey foreign KEY (orden_de_pago_id) references ordenes_de_pago (id)
) TABLESPACE pg_default;
