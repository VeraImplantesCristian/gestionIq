create table public.ordenes_de_pago (
  id bigint generated by default as identity not null,
  created_at timestamp with time zone null default now(),
  monto_total_general numeric not null,
  fecha_emision date not null,
  comprobante_object_key text null,
  notas text null,
  constraint ordenes_de_pago_pkey primary key (id)
) TABLESPACE pg_default;
