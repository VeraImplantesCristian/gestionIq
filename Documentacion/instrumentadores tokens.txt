create table public.instrumentador_tokens (
  id bigint generated by default as identity not null,
  instrumentador_dni text not null,
  token text not null,
  expires_at timestamp with time zone not null,
  created_at timestamp with time zone not null default now(),
  constraint instrumentador_tokens_pkey primary key (id),
  constraint instrumentador_tokens_token_key unique (token),
  constraint instrumentador_tokens_instrumentador_dni_fkey foreign KEY (instrumentador_dni) references instrumentadores (dni) on delete CASCADE
) TABLESPACE pg_default;
